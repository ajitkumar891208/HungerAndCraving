function validateForm() {
  const name = document.getElementById('textInput').value.trim();
  const email = document.getElementById('emailInput').value.trim();
  const age = document.getElementById('numberInput').value;
  const password = document.getElementById('passwordInput').value;
  const date = document.getElementById('dateInput').value;

  if (!/^[A-Za-z\s]+$/.test(name))
    return "‚ùå Invalid Name: Only letters allowed.";

  if (!/^[\w.-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$/.test(email))
    return "‚ùå Invalid Email Format.";

  if (age < 18 || age > 99)
    return "‚ùå Invalid Age: Must be between 18‚Äì99.";

  if (password.length < 6 || !/[0-9]/.test(password))
    return "‚ùå Weak Password: Min 6 chars and include a number.";

  const today = new Date();
  const inputDate = new Date(date);
  if (inputDate > today)
    return "‚ùå Invalid Date: Cannot be in the future.";

  return "‚úÖ All inputs are valid!";
}
